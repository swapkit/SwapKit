Exodus Wallet is a popular desktop and mobile wallet that supports multiple cryptocurrencies with a focus on user experience and design. SwapKit SDK v4 provides support for Exodus Wallet through their Passkeys integration and browser provider APIs.

## Supported Chains

SwapKit supports Exodus Wallet for these chains:

- **EVM Compatible**: All EVM chains (Ethereum, Arbitrum, Aurora, Avalanche, Base, BinanceSmartChain, Gnosis, Optimism, Polygon)
- **Bitcoin**: Native Bitcoin support with UTXO management
- **Solana**: SOL and SPL token support

## Installation and Setup

### 1. Install Exodus Wallet

Install Exodus Wallet from official sources:

- **Desktop**:
  - Download from [exodus.com](https:
  - Available for Windows, macOS, and Linux
- **Mobile App**:
  - iOS: [App Store](https:
  - Android: [Google Play](https:
- **Browser Extension**:
  - [Chrome Web Store](https:

### 2. Set Up Exodus Wallet

1. Create a new wallet or restore from seed phrase
2. Set up a secure password
3. Enable the cryptocurrencies you want to use
4. Backup your wallet securely

### 3. Requirements

- **Desktop**: Modern computer with internet connection
- **Mobile**: iOS 13+ or Android 8+
- **Browser**: Chrome, Firefox, Safari, Edge (for extension)
- **Web3 Support**: Ensure Web3 features are enabled in Exodus

### Connecting Exodus Wallet

#### Basic Connection

```ts twoslash
import { SwapKit, exodusWallet, Chain } from "@swapkit/sdk";
import { Wallet } from "@passkeys/core";

const swapKit = SwapKit({
  wallets: { ...exodusWallet },
});

async function connectExodusWallet() {
  try {
    console.log("Connecting to Exodus Wallet...");
    console.log("📱 This will open Exodus Wallet or prompt for connection");

    const exodusInstance = new Wallet();

    await swapKit.connectExodusWallet([Chain.Ethereum], exodusInstance);

    const address = swapKit.getAddress(Chain.Ethereum);
    const balance = await swapKit.getBalance(Chain.Ethereum, true);

    console.log("✅ Connected to Exodus Wallet:", address);
    console.log("ETH balance:", balance);

    return { success: true, address, balance };
  } catch (error) {
    console.error("Failed to connect Exodus Wallet:", error);
    return { success: false, error };
  }
}
```

#### Multi-Chain Connection

```ts twoslash
import { SwapKit, exodusWallet, Chain } from "@swapkit/sdk";
import { Wallet } from "@passkeys/core";

const swapKit = SwapKit({
  wallets: { ...exodusWallet },
});

async function connectMultipleChains() {
  const supportedChains = [
    Chain.Ethereum,
    Chain.Bitcoin,
    Chain.Solana,
    Chain.Polygon,
    Chain.Arbitrum,
    Chain.Avalanche,
  ];

  try {
    const exodusInstance = new Wallet();

    await swapKit.connectExodusWallet(supportedChains, exodusInstance);

    const addresses = {};
    for (const chain of supportedChains) {
      addresses[chain] = swapKit.getAddress(chain);
      console.log(`${chain} address:`, addresses[chain]);
    }

    const balances = await Promise.all(
      supportedChains.map(async (chain) => {
        try {
          const balance = await swapKit.getBalance(chain, true);
          return { chain, balance };
        } catch (error) {
          return { chain, balance: [], error: error.message };
        }
      })
    );

    return { success: true, addresses, balances };
  } catch (error) {
    console.error("Multi-chain connection failed:", error);
    return { success: false, error };
  }
}
```

## Security Considerations

### Best Practices

```ts twoslash
import { SwapKit, exodusWallet, Chain } from "@swapkit/sdk";
import { Wallet } from "@passkeys/core";

class ExodusWalletManager {
  private walletInstance: Wallet | null = null;
  private swapKit: any;

  constructor() {
    this.swapKit = SwapKit({ wallets: { ...exodusWallet } });
  }

  async initialize() {
    if (!this.walletInstance) {
      this.walletInstance = new Wallet();
    }
    return this.walletInstance;
  }

  async connect(chains: Chain[]) {
    const wallet = await this.initialize();

    try {
      await this.swapKit.connectExodusWallet(chains, wallet);
      return { success: true };
    } catch (error) {
      console.error("Connection failed:", error.message);
      throw new Error("Failed to connect to Exodus Wallet");
    }
  }

  async disconnect() {
    if (this.walletInstance) {
      try {
        await this.walletInstance.disconnect();
      } catch (error) {
        console.warn("Disconnect warning:", error.message);
      } finally {
        this.walletInstance = null;
        this.swapKit.disconnectAll();
      }
    }
  }
}

function validateAddress(address: string, chain: Chain): boolean {
  switch (chain) {
    case Chain.Ethereum:
    case Chain.Polygon:
    case Chain.Arbitrum:
    case Chain.Avalanche:
    case Chain.Base:
    case Chain.Optimism:
      return /^0x[a-fA-F0-9]{40}$/.test(address);

    case Chain.Bitcoin:
      return /^(1|3|bc1)[a-zA-Z0-9]{25,62}$/.test(address);

    case Chain.Solana:
      return /^[1-9A-HJ-NP-Za-km-z]{44}$/.test(address);

    default:
      return false;
  }
}

async function secureTransfer(chain: Chain, recipient: string, amount: number) {
  if (!validateAddress(recipient, chain)) {
    throw new Error(`Invalid ${chain} address format`);
  }

  if (amount <= 0) {
    throw new Error("Amount must be positive");
  }

  const walletManager = new ExodusWalletManager();
  await walletManager.connect([chain]);

  let assetValue: AssetValue;
  switch (chain) {
    case Chain.Ethereum:
      assetValue = AssetValue.fromChainOrSignature("ETH.ETH", amount);
      break;
    case Chain.Bitcoin:
      assetValue = AssetValue.fromChainOrSignature("BTC.BTC", amount);
      break;
    case Chain.Solana:
      assetValue = AssetValue.fromChainOrSignature("SOL.SOL", amount);
      break;
    default:
      throw new Error("Unsupported chain");
  }

  console.log("📋 Transaction Details:");
  console.log("Chain:", chain);
  console.log("Amount:", assetValue.toSignificant(6));
  console.log("Recipient:", recipient);
  console.log("⚠️ Please verify these details in Exodus Wallet");

  try {
    const swapKit = walletManager.swapKit;
    const txHash = await swapKit.transfer({
      assetValue,
      recipient,
    });

    return txHash;
  } catch (error) {
    console.error("Transaction failed:", error.message);
    throw new Error("Transaction failed. Please try again.");
  } finally {
    await walletManager.disconnect();
  }
}
```

## React Integration

```tsx twoslash
import React, { useState, useEffect } from "react";
import { SwapKit, exodusWallet, Chain, AssetValue } from "@swapkit/sdk";
import { Wallet } from "@passkeys/core";

interface ExodusWalletProps {
  onConnect: (addresses: Partial<Record<Chain, string>>) => void;
  onError: (error: Error) => void;
}

export function ExodusWallet({ onConnect, onError }: ExodusWalletProps) {
  const [isConnecting, setIsConnecting] = useState(false);
  const [isConnected, setIsConnected] = useState(false);
  const [addresses, setAddresses] = useState<Partial<Record<Chain, string>>>(
    {}
  );
  const [selectedChains, setSelectedChains] = useState<Chain[]>([
    Chain.Ethereum,
  ]);
  const [walletInstance, setWalletInstance] = useState<Wallet | null>(null);

  const swapKit = SwapKit({ wallets: { ...exodusWallet } });

  const isCompatible =
    typeof window !== "undefined" &&
    "credentials" in navigator &&
    "crypto" in window &&
    (window.location.protocol === "https:" ||
      window.location.hostname === "localhost");

  useEffect(() => {
    if (isCompatible && !walletInstance) {
      setWalletInstance(new Wallet());
    }
  }, [isCompatible]);

  const connectExodus = async () => {
    if (!walletInstance) {
      onError(new Error("Wallet instance not initialized"));
      return;
    }

    setIsConnecting(true);

    try {
      await swapKit.connectExodusWallet(selectedChains, walletInstance);

      const connectedAddresses: Partial<Record<Chain, string>> = {};
      for (const chain of selectedChains) {
        connectedAddresses[chain] = swapKit.getAddress(chain);
      }

      setAddresses(connectedAddresses);
      setIsConnected(true);
      onConnect(connectedAddresses);
    } catch (error) {
      onError(error as Error);
    } finally {
      setIsConnecting(false);
    }
  };

  const disconnect = async () => {
    if (walletInstance) {
      try {
        await walletInstance.disconnect();
      } catch (error) {
        console.warn("Disconnect warning:", error);
      }
    }
    swapKit.disconnectAll();
    setIsConnected(false);
    setAddresses({});
  };

  const toggleChain = (chain: Chain) => {
    if (isConnected) return;

    setSelectedChains((prev) =>
      prev.includes(chain) ? prev.filter((c) => c !== chain) : [...prev, chain]
    );
  };

  const supportedChains = [
    { chain: Chain.Ethereum, name: "Ethereum" },
    { chain: Chain.Bitcoin, name: "Bitcoin" },
    { chain: Chain.Solana, name: "Solana" },
    { chain: Chain.Polygon, name: "Polygon" },
    { chain: Chain.Arbitrum, name: "Arbitrum" },
    { chain: Chain.Avalanche, name: "Avalanche" },
  ];

  if (!isCompatible) {
    return (
      <div>
        <h3>Exodus Wallet</h3>
        <p>⚠️ Browser compatibility issues</p>
        <p>Requirements:</p>
        <ul>
          <li>Modern browser with WebCrypto API</li>
          <li>HTTPS connection (or localhost)</li>
          <li>Passkeys/WebAuthn support</li>
        </ul>
      </div>
    );
  }

  return (
    <div>
      <h3>Exodus Wallet</h3>
      <p>🔐 Secure multi-chain wallet with passkeys</p>

      {!isConnected ? (
        <div>
          <h4>Select Chains:</h4>
          {supportedChains.map(({ chain, name }) => (
            <label key={chain}>
              <input
                type="checkbox"
                checked={selectedChains.includes(chain)}
                onChange={() => toggleChain(chain)}
                disabled={isConnecting}
              />
              {name}
            </label>
          ))}

          <button
            onClick={connectExodus}
            disabled={
              isConnecting || selectedChains.length === 0 || !walletInstance
            }
          >
            {isConnecting ? "Connecting..." : "Connect Exodus Wallet"}
          </button>

          {isConnecting && <p>📱 Follow the prompts in Exodus Wallet</p>}
        </div>
      ) : (
        <div>
          <h4>Connected Addresses:</h4>
          {Object.entries(addresses).map(([chain, address]) => (
            <div key={chain}>
              <strong>{chain}:</strong> {address}
            </div>
          ))}
          <button onClick={disconnect}>Disconnect</button>
        </div>
      )}
    </div>
  );
}
```
